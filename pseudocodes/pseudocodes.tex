	\begin{algorithm}
		\caption{K-Means}
		\label{array-sum}
		\begin{algorithmic}[1]
			\Function{K-Means}{$k$, $dataset$, $distance\_method$, $tolerance$, $max\_iterations$}
			\State $centroid \leftarrow$ Initialize\_centoids($k$)
			\State $changed \leftarrow True$
			\State $iteration \leftarrow 0$
			\While {$changed$}
			    \State $iteration \leftarrow iteration + 1$
			    \State $classification \leftarrow$ classifies\_points($dataset$, $distance\_method$)
    			\State $previous \leftarrow$ copy($centroids$)
    			\State $centroids \leftarrow$ update\_centroids($centroids$, $classification$)
    			\If{$iteration = max\_iterations$}
    			    \State $changed \leftarrow False$
    			\ElsIf{$centroids = previous$}
    			    \State $changed \leftarrow False$
    			\ElsIf{stop\_threshold($centroids$, $previous$, $tolerance$)}
    			    \State $changed \leftarrow False$
    			\EndIf
			\EndWhile
			\State Return $centroids$, $classification$
			\EndFunction
		\end{algorithmic}
	\end{algorithm}
